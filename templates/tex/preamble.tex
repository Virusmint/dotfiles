% Defines lecture command with position arguments: lecture #, date, title

  \iffinalversion
    % Do nothing
  \else
    \ifthenelse{\isempty{#3}}{%
        \def\@lecture{Lect. #1}%
    }{
        \def\@lecture{Lect. #1: #3}%

    \subsection*{\@lecture}
    \marginpar{\small\textsf{\mbox{#2}}}
  \fi
}
